rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Shipments: allow public reads, but only admins can write/create/update/delete
    match /shipments/{shipmentId} {
      allow read: if true;
      allow create, update, delete: if request.auth != null && request.auth.token.admin == true;
    }

    // Subscribers: allow public reads; allow anyone to create a subscriber,
    // but only admins can modify or delete. Adjust create rule if you require auth.
    match /subscribers/{subId} {
      allow read: if true;
      allow create: if true; // or: if request.auth != null;
      allow update, delete: if request.auth != null && request.auth.token.admin == true;
    }

    // Fallback: allow reads but block writes by default
    match /{document=**} {
      allow read: if true;
      allow write: if false;
    }
  }
}
